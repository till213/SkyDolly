        include/Model/FlightCondition.h src/FlightCondition.cpp

        include/Model/PositionData.h src/PositionData.cpp

        include/Model/Engine.h src/Engine.cpp

    PUBLIC
        Qt6::Core
        Qt6::Sql
    PRIVATE
        Sky::Kernel
)

    PROPERTIES
        VERSION ${PROJECT_VERSION}
        SOVERSION ${PROJECT_VERSION_MAJOR}
